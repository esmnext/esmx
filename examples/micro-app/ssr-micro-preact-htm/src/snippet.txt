import { html } from 'htm/preact'
import { useState } from 'preact/hooks'

export function Counter() {
  const [count, setCount] = useState(0)
  return html`
    <p>Count: ${count}</p>
    <button onClick=${() => setCount(c => c + 1)}>+</button>
    <button onClick=${() => setCount(c => c - 1)}>âˆ’</button>
  `
}
